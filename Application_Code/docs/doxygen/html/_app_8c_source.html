190static void App_DefaultOnOffBtnHandler(void)
191{
192 if(App_u8SystemState == SYSTEM_RUNNING)
193 {
194 App_u8SystemState = SYSTEM_STOPPED;
195 App_u8SettingModeState = SETTING_MODE_DIS;
196 App_u16SetModeTimerCounter = 0;
197 App_u8TempTimerCounter = 0;
198 App_TimerStop();
199 DSSD_voidDisableAll();
200 Heater_voidTurnOff(HEATER_RELAY_PIN);
201 Cooler_voidTurnOff(COOLER_RELAY_PIN);
202 LED_voidTurnOff(LED_C_H);
203 PushBtn_voidDisable(BTN_UP_CTRL);
204 PushBtn_voidDisable(BTN_DOWN_CTRL);
205 }
206 else
207 {
208 App_u8SystemState = SYSTEM_RUNNING;
209 App_TimerBegin();
210 PushBtn_voidEnable(BTN_UP_CTRL);
211 PushBtn_voidEnable(BTN_DOWN_CTRL);
212 }

289static void App_DefaultTimerHandler(void)
290{
291 App_u8TempTimerCounter++;
292 if(App_u8TempTimerCounter % 10 == 0)
293 {
294 App_u8CurrentTemp = SensorLM35_u16ReadTemp(LM35_PIN);
295 App_u8TempAverage[(App_u8TempTimerCounter / 10) - 1] = App_u8CurrentTemp;
296 for(u8 i = 0; i < 10; i++)
297 {
298 App_u16TempAverageCalc += App_u8TempAverage[i];
299 }
300 if(((f32)(App_u16TempAverageCalc / 10)) > ((f32)(App_u8TempSetPoint + TEMP_TOLERANCE)))
301 {
302 Cooler_voidTurnOn(COOLER_RELAY_PIN);
303 Heater_voidTurnOff(HEATER_RELAY_PIN);
304 LED_voidTurnOn(LED_C_H);
305 }
306 else if(((f32)(App_u16TempAverageCalc / 10)) < ((f32)(App_u8TempSetPoint - TEMP_TOLERANCE)))
307 {
308 Heater_voidTurnOn(HEATER_RELAY_PIN);
309 Cooler_voidTurnOff(COOLER_RELAY_PIN);
310 if(App_u8TempTimerCounter >= 100) LED_voidToggle(LED_C_H);
311 }
312 else
313 {
314 Heater_voidTurnOff(HEATER_RELAY_PIN);
315 Cooler_voidTurnOff(COOLER_RELAY_PIN);
316 LED_voidTurnOff(LED_C_H);
317 }
318 App_u16TempAverageCalc = 0;
319 }
320 if(App_u8TempTimerCounter >= 100)
321 {
322 App_u8TempTimerCounter = 0;
323 }
324
325 if(App_u8SettingModeState == SETTING_MODE_EN)
326 {
327 App_u16SetModeTimerCounter++;
328 if(App_u16SetModeTimerCounter % 100 == 0)
329 {
330 if(DSSD_u8Status() == DSSD_ON)
331 {
332 DSSD_voidDisableAll();
333 }
334
335 else if(DSSD_u8Status() == DSSD_OFF)
336 {
337 DSSD_voidEN1();
338 }
339 }
340 if(DSSD_u8Status() == DSSD_ON)
341 {
342 DSSD_voidDisplay(App_u8TempSetPoint, App_u8SevenSegDigit);
343 }
344 if(App_u16SetModeTimerCounter >= 500)
345 {
346 App_u8SettingModeState = SETTING_MODE_DIS;
347 App_u16SetModeTimerCounter = 0;
348 EEPROM_voidWriteByte(App_u8TempSetPoint, 0, 0);
349 }
350 }
351 else if(App_u8SettingModeState == SETTING_MODE_DIS)
352 {
353 DSSD_voidDisplay(App_u8CurrentTemp, App_u8SevenSegDigit);
354 }
355 App_u8SevenSegDigit = App_u8SevenSegDigit == DSSD_DISPLAY_ONES ? DSSD_DISPLAY_TENS : DSSD_DISPLAY_ONES;
356}

415void App_SystemInit(void)
416{
417 GIE_voidEnableGlobalINT();
418
419 if(App_ptrOnOffBtnHandler == NULL) App_ptrOnOffBtnHandler = App_DefaultOnOffBtnHandler;
420 if(App_ptrUpBtnHandler == NULL) App_ptrUpBtnHandler = App_DefaultUpBtnHandler;
421 if(App_ptrDownBtnHandler == NULL) App_ptrDownBtnHandler = App_DefaultDownBtnHandler;
422 if(App_ptrTimerHandler == NULL) App_ptrTimerHandler = App_DefaultTimerHandler;
423
424 PushBtn_voidInit(BTN_ON_OFF, PUSH_BTN_INTERRUPT_EN, App_ptrOnOffBtnHandler);
425 PushBtn_voidInit(BTN_UP_CTRL, PUSH_BTN_INTERRUPT_EN, App_ptrUpBtnHandler);
426 PushBtn_voidInit(BTN_DOWN_CTRL, PUSH_BTN_INTERRUPT_EN, App_ptrDownBtnHandler);
427
428 App_TimerInit();
429 DSSD_voidInit();
430 SensorLM35_voidInit();
431
432 App_u8CurrentTemp = SensorLM35_u16ReadTemp(LM35_PIN);
433 for(u8 i = 0; i < 10; i++)
434 {
435 App_u8TempAverage[i] = App_u8CurrentTemp;
436 }
437
438 Heater_voidInit(HEATER_RELAY_PIN);
439 LED_voidInit(LED_C_H);
440 Cooler_voidInit(COOLER_RELAY_PIN);
441 EEPROM_voidInit();
442 _delay_ms(1);
443
444 EEPROM_u8ReadByte(0, 0, &App_u8TempSetPoint);
445 _delay_ms(1);
446 if(App_u8TempSetPoint == 255)
447 {
448 App_u8TempSetPoint = TEMP_INITIAL_SET_POINT;
449 EEPROM_voidWriteByte(TEMP_INITIAL_SET_POINT, 0, 0);
450 _delay_ms(1);
451 }
452}
