39void ADC_voidInit() {
40 /* VREF setting */
41 WRT_2BITS(ADC_ADMUX, ADC_ADMUX_REFS0, ADC_VOLT_REF);
42
43 /* ADC Right Adjust */
44 CLR_BIT(ADC_ADMUX, ADC_ADMUX_ADLAR);
45
46 /* ADC Prescaler */
47 WRT_3BITS(ADC_ADCSRA, ADC_ADCSRA_ADPS0, ADC_PRESCALER);
48
49 /* ADC Auto Trigger State */
50 WRT_BIT(ADC_ADCSRA, ADC_ADCSRA_ADATE, ADC_AUTO_TRIGGER_STATE);
51
52 /* ADC Interrupt Enable/Disable */
53 #if ADC_MODE == ADC_INTERRUPT_MODE
54 SET_BIT(ADC_ADCSRA, ADC_ADCSRA_ADIE);
55 #elif ADC_MODE == ADC_POLLING_MODE
56 CLR_BIT(ADC_ADCSRA, ADC_ADCSRA_ADIE);
57 #endif
58
59 /* ADC Auto Trigger Source */
60 WRT_3BITS(ADC_SFIOR, ADC_SFIOR_ADTS0, ADC_TRIG_SRC);
61
62 /* ADC Enable */
63 SET_BIT(ADC_ADCSRA, ADC_ADCSRA_ADEN);
64}

84 /* ADC Start Conversion */
