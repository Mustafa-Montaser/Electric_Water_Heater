38void IIC_voidInit(void) {
39
40 /* F_SCL Setting */
41 #if IIC_OPERATION_MODE == IIC_MASTER_MODE
42 if(((F_OSC_HZ - (16 * IIC_F_SCL_HZ)) / (2 * IIC_F_SCL_HZ * 1)) <= 255) {
43 IIC_TWBR = (F_OSC_HZ - (16 * IIC_F_SCL_HZ)) / (2 * IIC_F_SCL_HZ * 1);
44 WRT_2BITS(IIC_TWSR, IIC_TWSR_TWPS0, IIC_BIT_RATE_PRESCALER_1);

46 } else if(((F_OSC_HZ - (16 * IIC_F_SCL_HZ)) / (2 * IIC_F_SCL_HZ * 4)) <= 255) {
47 IIC_TWBR = (F_OSC_HZ - (16 * IIC_F_SCL_HZ)) / (2 * IIC_F_SCL_HZ * 4);
48 WRT_2BITS(IIC_TWSR, IIC_TWSR_TWPS0, IIC_BIT_RATE_PRESCALER_4);
49
50 } else if(((F_OSC_HZ - (16 * IIC_F_SCL_HZ)) / (2 * IIC_F_SCL_HZ * 16)) <= 255) {
51 IIC_TWBR = (F_OSC_HZ - (16 * IIC_F_SCL_HZ)) / (2 * IIC_F_SCL_HZ * 16);
52 WRT_2BITS(IIC_TWSR, IIC_TWSR_TWPS0, IIC_BIT_RATE_PRESCALER_16);
53
54 } else {
55 IIC_TWBR = (F_OSC_HZ - (16 * IIC_F_SCL_HZ)) / (2 * IIC_F_SCL_HZ * 64);
56 WRT_2BITS(IIC_TWSR, IIC_TWSR_TWPS0, IIC_BIT_RATE_PRESCALER_64);

58 #endif
59

61 SET_BIT(IIC_TWCR, IIC_TWCR_TWEA);
62
63 /* Write Slave Address [For This Device] */
64 IIC_TWAR = IIC_SLAVE_ADDRESS << 1;
65
66 /* Listen To General Call State */
67 WRT_BIT(IIC_TWAR, IIC_TWAR_TWGCE, IIC_LISTEN_GENERAL_CALL);
68
69 /* Clear The Flag ( with [SET] ) */
70// SET_BIT(IIC_TWCR, IIC_TWCR_TWINT);
71
72 /* IIC Enable */
73 SET_BIT(IIC_TWCR, IIC_TWCR_TWEN);

79IIC_State_t IIC_enumStartCondition(void) {
80
81 IIC_State_t local_enumState = IIC_NOK;
82
83 /* Start Condition */
84 SET_BIT(IIC_TWCR, IIC_TWCR_TWSTA);
85
86 /* Clear The Flag ( with [SET] ) */
87 SET_BIT(IIC_TWCR, IIC_TWCR_TWINT);
88
89 /* Waiting till Job Is Done [Start Condition] */
90 u32 local_u32Loop = 0;
91 while(GET_BIT(IIC_TWCR, IIC_TWCR_TWINT) == 0 && local_u32Loop < IIC_TIMEOUT) {
92 local_u32Loop++;
93 }
94
95 if(local_u32Loop >= IIC_TIMEOUT) {
96 local_enumState = IIC_TIMEOUT_ERR;
97 } else {
98 if((IIC_TWSR & 0xF8) == 0x08)
99 local_enumState = IIC_OK;
100 else
101 local_enumState = IIC_START_COND_ERR;
102 }
103
104 /* Clear TWSTA Bit After Start Condition */
105// CLR_BIT(IIC_TWCR, IIC_TWCR_TWSTA);
106
107 return local_enumState;
108

140IIC_State_t IIC_enumSendSlaveAddressWithWrite(u8 local_u8SlaveAddress) {
141
142 IIC_State_t local_enumState = IIC_NOK;
143
144 /* Writing Slave Address + [WRITE] */
145 IIC_TWDR = local_u8SlaveAddress << 1;
146
147 /* Clear Required Bits in TWCR */
148 CLR_BIT(IIC_TWCR, IIC_TWCR_TWSTA);
149 CLR_BIT(IIC_TWCR, IIC_TWCR_TWSTO);
150
151 /* Clear The Flag ( with [SET] ) */
152 SET_BIT(IIC_TWCR, IIC_TWCR_TWINT);
153
154 /* Waiting till Job Is Done [Send Slave Address + Write] */
155 u32 local_u32Loop = 0;
156 while(GET_BIT(IIC_TWCR, IIC_TWCR_TWINT) == 0 && local_u32Loop < IIC_TIMEOUT) {
157 local_u32Loop++;
158 }
159
160 if(local_u32Loop >= IIC_TIMEOUT) {
161 local_enumState = IIC_TIMEOUT_ERR;
162 } else {
163 if((IIC_TWSR & 0xF8) == 0x18)
164 local_enumState = IIC_OK;
165 else if((IIC_TWSR & 0xF8) == 0x20)
166 local_enumState = IIC_SLA_ADD_WRT_NACK;
167 else if((IIC_TWSR & 0xF8) == 0x38)
168 local_enumState = IIC_ARBITRATION_LOST;
169 else
170 local_enumState = IIC_SLA_ADD_WRT_ERR;
171 }
