69 void Timer0_voidSetTime(u32 copy_u8Time) {
70 /* Calc. Tick Time = (1 * prescaler) / F_OSC */
71 u32 local_u32TickTime = 0;
72 #if TIMER0_PRESCALER == TIMER0_NO_PRESCALER
73 local_u32TickTime = 1 / TIMER0_F_OSC;
74 #elif TIMER0_PRESCALER == TIMER0_PRESCALER_8
75 local_u32TickTime = 8 / TIMER0_F_OSC;
76 #elif TIMER0_PRESCALER == TIMER0_PRESCALER_64
77 local_u32TickTime = 64 / TIMER0_F_OSC;
78 #elif TIMER0_PRESCALER == TIMER0_PRESCALER_256
79 local_u32TickTime = 256 / TIMER0_F_OSC;
80 #elif TIMER0_PRESCALER == TIMER0_PRESCALER_1024
81 local_u32TickTime = 1024 / TIMER0_F_OSC;
82 #endif
83
84 #if TIMER0_WAVEFORM_MODE == TIMER0_WAVEFORM_NORMAL_MODE
85 u32 local_u32TotalNumOfTicks = (copy_u8Time * 1000) / local_u32TickTime;
86 Timer0_u32NumOfOverflows = local_u32TotalNumOfTicks / 256;
87 Timer0_u8NumOfReminderTicks = local_u32TotalNumOfTicks % 256;
88 if(Timer0_u8NumOfReminderTicks != 0) {
89 TIMER0_TCNT0 = 256 - Timer0_u8NumOfReminderTicks;
90 Timer0_u32NumOfOverflows++;
91 }
92
93 #elif TIMER0_WAVEFORM_MODE == TIMER0_WAVEFORM_CTC_MODE
94 u32 local_u32TotalNumOfTicks = (copy_u8Time * 1000) / local_u32TickTime;
95 u8 local_u8DivisionFactor = 255;
96 while(local_u32TotalNumOfTicks % local_u8DivisionFactor) {
97 local_u8DivisionFactor --;
98 }
99 Timer0_u8NumOfCompMatch = local_u32TotalNumOfTicks / local_u8DivisionFactor;
100 TIMER0_OCR0 = local_u8DivisionFactor - 1;
101 #endif
102 }
103
104#endif /* Normal and CTC mode */
