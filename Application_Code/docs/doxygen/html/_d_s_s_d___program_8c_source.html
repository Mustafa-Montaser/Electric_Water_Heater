217u8 DSSD_u8Status()
218{
220 u8 local_u8state = 0;
221 if(GET_BIT(DSSD_CTRL_REG, DSSD_EN1 % 10) && GET_BIT(DSSD_CTRL_REG, DSSD_EN2 % 10))
222 {
223 local_u8state = DSSD_OFF;
224 }
225 else
226 {
227 local_u8state = DSSD_ON;
228 }
229 return local_u8state;
230}
